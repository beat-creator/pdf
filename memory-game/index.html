async function handleTileClick(id) { if (isWaiting) return; if (id === sequence[playerIndex]) { flashTile(id, 150); playerIndex++; if (playerIndex === sequence.length) { isWaiting = true; if (totalAttempts > highScore) { highScore = totalAttempts; localStorage.setItem('memoryGameBest', highScore); bestDisplay.innerText = `BEST LEVEL: ${highScore}`; } setTimeout(nextLevel, 800); } } else { isWaiting = true; await flashTile(id, 500, 'error'); statusText.innerText = "ミス！終了です"; statusText.style.color = "var(--danger)"; // 修正点 await sleep(1000); startScreen.classList.remove('hidden'); } } // nextLevel関数内の色指定も修正 async function nextLevel() { isWaiting = true; playerIndex = 0; totalAttempts++; const memoryCount = Math.floor((totalAttempts - 1) / 3) + 2; const isFast = (totalAttempts % 3 === 0); const displaySpeed = isFast ? 200 : 500; sequence = []; for (let i = 0; i < memoryCount; i++) { sequence.push(Math.floor(Math.random() * 16)); } levelDisplay.innerText = totalAttempts; statusText.innerText = isFast ? "スピードアップ！" : "覚えてください..."; statusText.style.color = isFast ? "var(--warning)" : "var(--muted)"; // 修正点 await sleep(1000); for (const id of sequence) { await flashTile(id, displaySpeed); await sleep(isFast ? 100 : 250); } statusText.innerText = "再現してください"; statusText.style.color = "var(--accent)"; // 修正点：文字列にする isWaiting = false; }
